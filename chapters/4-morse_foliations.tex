\chapter{Morse Foliations with Controlled Area of Fibers}\label{ch:morse-foliation}

The goal of this chapter is to develop technical tools for constructing \emph{nested} sweepouts from arbitrary sweepouts. The central problem is the following: given a sweepout $\{\Gamma_t\}$ that may not be nested (i.e., $\Omega_s \not\subset \Omega_t$ for $s < t$), how can we deform it into a nested sweepout while controlling the area of the slices?

Our strategy is to work locally: we construct Morse functions on regions of the doubled manifold $\widetilde{M}$ whose level sets provide the desired nested structure. The key technical results are Lemma~\ref{lem:morse-foliation-1} and Lemma~\ref{lem:morse-foliation-non-compact}, which show that given any Morse function $f$ and a target hypersurface $\Sigma$, we can perturb $f$ to a new Morse function $g$ having $\Sigma$ as a level set, while preserving area bounds. These local constructions are then glued together in Chapter~\ref{ch:nested-sweepout} to produce global nested sweepouts.

\section{Gluing Morse Foliations}
\begin{lemma}\label{lem:morse-foliation-1}
    Let $N^{n + 1} \subset \Tilde{M}^{n + 1}$ be a compact submanifold with non empty boundary. Let $f: N \to [a, b]$ be a $\partial$-transverse Morse function. Let $\Sigma^n$ be a closed submanifold of $\partial N$.

    For every $\eps > 0$, there exists a Morse function $g: N \to [a, b]$, such that the following holds:
    \begin{enumerate}
        \item[(1)] $g^{-1}(b) = \Sigma$;
        \item[(2)] $f^{-1}([a, t)) \subset N_{\eps/2}(g^{-1}([a, t))) \subset N_\eps(f^{-1}([a, t)))$;
        \item[(3)] $\haus^n(g^{-1}(t)\cap M) \leq \haus^n(\partial f^{-1}([a, t])\cap M) + \eps$;
        \item[(4)] If $\dist(x, f(\Sigma)) > \eps$ then $f^{-1}(x) = g^{-1}(x)$. 
    \end{enumerate}
\end{lemma}
\begin{remark}
    (3) is trivial if $N \subset \Tilde{M}\backslash M$. In fact, we always consider the situation when $N\cap M$ is a relative open subset of $M$.
\end{remark}
We will also need a slightly different version of this lemma for a non-compact submanifold $N$.
\begin{lemma}\label{lem:morse-foliation-non-compact}
    Let $N \subset \Tilde{M}$ be a not necessarily compact submanifold with non-empty boundary and $f: N \to (-\infty, b]$ be a proper Morse function, which is $\partial$-transverse. Let $\Sigma$ be a compact submanifold of $\partial N$.

    For every $\eps > 0$ there exists a Morse function $g: N \to (-\infty, b]$, such that the following holds:
    \begin{enumerate}
        \item[(1)] $g^{-1}(b) = \Sigma$;
        \item[(2)] $f^{-1}((-\infty, t)) \subset N_{\eps/2}(g^{-1}((-\infty, t))) \subset N_\eps(f^{-1}((-\infty, t)))$;
        \item[(3)] $\haus^n(g^{-1}(t)\cap M) \leq \haus^n(\partial f^{-1}((-\infty, t])\cap M) + \eps$;
        \item[(4)] If $\dist(x, f(\Sigma)) > \eps$ then $f^{-1}(x) = g^{-1}(x)$. 
    \end{enumerate}
\end{lemma}
\section[Gluing Morse Foliations on Separated Manifolds]{Gluing Morse Foliations on a Manifold Separated by a hypersurface transverse to the boundary}
\begin{lemma}\label{lem:morse-separate}
    Let $N^{n + 1}\subset \Tilde{M}^{n + 1}$ be a manifold with compact boundary $\partial N$ and $\Sigma$ be a hypersurface with $\partial\Sigma \subset \partial N$ and such that $\Sigma$ intersects $\partial N$ transversally and separates $N$ into two disjoint regions $N\backslash \Sigma = V_1 \sqcup V_2$. 

    For every $\eps > 0$, there exist open sets with smooth boundary $\Omega_1$ and $\Omega_2$ and a Morse function $g: cl(N\backslash(\Omega_1\cup\Omega_2)) \to [0, 1]$, such that the following holds:
    \begin{enumerate}
        \item[(1)] $\Omega_1$ is a relative inward $\eps$-perturbation of $V_1$; $\Omega_2$ is a relative inward $\eps$-perturbation of $V_2$. 
        \item[(2)] $g^{-1}(0) = \partial\Omega_1\cup\partial\Omega_2$ and $g^{-1}(1) = \partial N$;
        \item[(3)] $\haus^n(g^{-1}(t)\cap M) \leq \haus^n(\partial N\cap M) + 2\haus^n(\Sigma\cap M) + \eps$. 
    \end{enumerate}
    \begin{proof}
        Let $f: \Tilde{M} \to \R$ be a Morse function such that there exists some sufficiently small $0 < \delta_1 < \eps$, 
        \begin{equation*}
            f(x) = sdist(x, \partial N)
        \end{equation*}
        for $x \in N_{\delta_1}(\partial N)$. The sign is chosen so that $f(x) \geq 0$ when $x \in N$. Without any loss of generality, we may assume that $f$ has no critical points in the $\eps$-neighborhood of $\Sigma$, the restriction of $f$ to $\Sigma$ is also Morse and
        \begin{equation*}
            \haus^n(f^{-1}(t)\cap M) \leq \haus^n(\partial N\cap M) + \eps/10
        \end{equation*}
        for $t \in [0, \delta_1]$. Pick $\delta_2 \in (0, \delta_1)$ so that for $t \in (0, \delta_2)$,
        \begin{equation*}
            \haus^n((f^{-1}(t)\cap N_{\delta_2}(\Sigma))\cap M) < \eps/10
        \end{equation*}
        and
        \begin{equation*}
            \haus^n(\partial N_{\delta_2}(\Sigma)\cap M) \leq 2\haus^n(\Sigma\cap M) + \eps/10.
        \end{equation*}

        Let $\Omega_i$ denote an relative inward $\delta_2/2$-perturbation of $V_i$, i.e.
        \begin{enumerate}
            \item[(a)] $V_i\backslash N_{\delta_2/2}(\partial V_i) \subset \Omega_i \subsetneq V_i$;
            \item[(b)] there exists a nested family of open sets $\br{\Xi_t^i}_{t \in [0, 1]}$ and corresponding smooth isotopies $\Gamma_t^i = \partial\Xi_t^i$, such that $\Gamma_1^i$ is a smoothing of $\partial V_i$ and 
            \begin{equation*}
                \haus^n(\Gamma_t^i\cap M) \leq \haus^n(\partial V_i\cap M) + \delta_2/2
            \end{equation*}
            for all $t \in [0, 1]$. 
        \end{enumerate}
        
        We can make an arbitrarily small perturbation of $f$ by an ambient diffeomorphism so that $f$ is a Morse function on $\partial\Omega_1\cup\partial\Omega_2$ and for $t \in [0, \delta_2]$ each leaf of the foliation $\Sigma_t = (f^{-1}(t)\cap N)\backslash(\Omega_1\cup\Omega_2)$ is a graph over some region $U_t\subset \partial N$ with
         \begin{figure}[h]
            \centering
            \includegraphics[width=0.8\textwidth]{figures/IMG_2671.PNG}
            \caption{Gluing two submanifolds using a Morse function}
            \label{fig:2671}
        \end{figure}

        \begin{equation*}
            \haus^n(\Sigma_t\cap M) \leq \haus^n(U_t\cap M) + \eps/5
        \end{equation*}

        It follows that
        \begin{equation*}
            \haus^n(\partial f|_{\Tilde{M}\backslash(\Omega_1\cup\Omega_2)}^{-1}((-\infty, t))\cap M) \leq 2\haus^n(\Sigma\cap M) + \haus^n(\partial N\cap M) + \eps/2.
        \end{equation*}
        We can now apply Lemma~\ref{lem:morse-foliation-non-compact} to $f|_{\Tilde{M}\backslash(\Omega_1\cup\Omega_2)}$ to obtain the desired function $g$ and the corresponding foliation, i.e.  
        \begin{equation*}
            \haus^n(g^{-1}(t)\cap M) \leq 2\haus^n(\Sigma\cap M) + \haus^n(\partial N\cap M) + \eps.
        \end{equation*}
    \end{proof}
\end{lemma}
